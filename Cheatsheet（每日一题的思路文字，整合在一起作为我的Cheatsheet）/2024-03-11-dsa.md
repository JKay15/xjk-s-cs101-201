def reverse_dialogue(script):

stack = []

dialogue = ''

  

for char in script:

if char == '(':

stack.append(dialogue)

dialogue = ''

elif char == ')':

dialogue = stack.pop() + dialogue[::-1]

else:

dialogue += char

  

return dialogue

  

# 测试样例
